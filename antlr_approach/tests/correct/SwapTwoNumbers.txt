#include <iostream>
using namespace std;

int main()
{
    int a = 5;
    int b = 10;
    int temp;

    cout << "Before swapping.";
    cout << "a = " << a << ", b = " << b << endl;

    temp = a;
    a = b;
    b = temp;

    cout << "\nAfter swapping." << endl;
    cout << "a = " << a << ", b = " << b << endl;

    return 0;
};
